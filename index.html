vlet map; let mapReady = false; // 🔥 安全初始化（等 DOM 完全好） window.addEventListener("load", ()=>{ setTimeout(()=>{ map = L.map('map').setView([23.7,121],7); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); // ✅ 關鍵修復 setTimeout(()=>{ map.invalidateSize(); mapReady = true; },500); },300); }); // 資料 const cities=[ {name:"台北",lat:25.03,lng:121.56}, {name:"台中",lat:24.15,lng:120.67}, {name:"高雄",lat:22.63,lng:120.30} ]; const foods=["雞排","飲料","火鍋","拉麵","甜點"]; // 🔥 主結果（穩定版） function showResult(title,text){ const main=document.getElementById("mainResult"); main.innerHTML="🔄 分析中..."; setTimeout(()=>{ const score = Math.floor(Math.random()*40)+60; main.innerHTML=`
${title}
${score}分
${text}
`; },300); } // 🔥 地圖事件（降頻版本，避免炸） function spawnMapEvent(){ if(!mapReady) return; const city=cities[Math.floor(Math.random()*cities.length)]; const food=foods[Math.floor(Math.random()*foods.length)]; const marker=L.circleMarker([city.lat,city.lng],{ radius:8, color:"orange", fillOpacity:0.7 }).addTo(map); marker.on("click",()=>{ showResult(city.name+"｜"+food,"市場關注上升"); }); setTimeout(()=>{ map.removeLayer(marker); },4000); } // ❗ 降頻（關鍵） setInterval(spawnMapEvent,4000); // 🔍 搜尋（簡化穩定版） function searchKeyword(){ const input=document.getElementById("keywordInput").value; if(!input) return; showResult(input,"市場需求上升"); } // Enter document.getElementById("keywordInput") .addEventListener("keypress",function(e){ if(e.key==="Enter") searchKeyword(); });
