// 🔥 升級 radarState radarState.sources = ["Threads","PTT","Dcard","IG"]; radarState.searchHistory = []; // 🔥 安全 triggerScan（相容舊版） window.triggerScan = function(manualTopic=null){ const input = document.getElementById('keywordInput').value; const topic = manualTopic || input || radarState.keywords[Math.floor(Math.random()*radarState.keywords.length)]; const score = Math.floor(Math.random()*40)+60; const source = radarState.sources[Math.floor(Math.random()*radarState.sources.length)]; const city = radarState.cities[Math.floor(Math.random()*radarState.cities.length)]; // 🔥 搜尋紀錄 if(input){ if(!radarState.searchHistory.includes(input)){ radarState.searchHistory.unshift(input); if(radarState.searchHistory.length > 3){ radarState.searchHistory.pop(); } } } radarState.pings.push({ x: city.x, y: city.y, radius:0, alpha:1 }); updateDOM(topic, score, source, city.name); document.getElementById('keywordInput').value=""; } // 🔥 升級 updateDOM（但保留相容） function updateDOM(topic, score, source="系統", cityName="未知"){ let trendIcon="📊"; let trendColor="#F59E0B"; let text="市場穩定中"; if(score>=90){ trendIcon="🔥"; trendColor="green"; text="需求爆發，建議立即切入"; }else if(score<70){ trendIcon="📉"; trendColor="red"; text="熱度下降，建議觀察"; } document.getElementById('intercept-content').innerHTML=`
📡 ${source}｜${cityName}
${topic}
${score} ${trendIcon}
${text} `; // 🔥 排行榜（簡化安全版） radarState.rankings[topic]=score; const sorted = Object.entries(radarState.rankings) .sort((a,b)=>b[1]-a[1]) .slice(0,5); let html=""; sorted.forEach((item,i)=>{ html+=`
${["🥇","🥈","🥉"][i]||""} ${item[0]} (${item[1]})
`; }); // 🔥 加搜尋紀錄 if(radarState.searchHistory.length>0){ html+=`
熱搜：${radarState.searchHistory.join("、")}
`; } document.getElementById("ranking-content").innerHTML=html; }
